dashboard/src/app/logs/page.tsx
"use client";
import Shell from "@/components/Shell";
import { useEffect, useMemo, useState } from "react";

export default function LogsPage() {
  const [events, setEvents] = useState<any[]>([]);

  useEffect(() => {
    const es = new EventSource("/api/stream");
    es.onmessage = (msg) => {
      const ev = JSON.parse(msg.data);
      setEvents((prev) => [...prev.slice(-500), ev]);
    };
    return () => es.close();
  }, []);

  const lines = useMemo(() => events.slice().reverse(), [events]);

  return (
    <Shell>
      <aside className="border-r border-zinc-800 p-4 space-y-2">
        <div className="text-lg font-semibold">Logs</div>
        <a className="text-sm hover:underline" href="/">← Back</a>
      </aside>

      <main className="p-4">
        <div className="font-semibold mb-2">Event Stream</div>
        <div className="bg-zinc-900 border border-zinc-800 rounded-xl p-3 h-[85vh] overflow-auto text-xs">
          {lines.map((e) => (
            <div key={e.id} className="py-1 border-b border-zinc-800/50">
              <div className="text-zinc-400">
                {new Date(e.created_at).toLocaleString()} • <span className="text-zinc-200">{e.kind}</span>
              </div>
              <div className="text-zinc-100">{e.message}</div>
            </div>
          ))}
        </div>
      </main>

      <section className="border-l border-zinc-800 p-4">
        <div className="font-semibold">Controls (MVP)</div>
        <p className="text-xs text-zinc-400 mt-2">
          Próximo: pause/resume global e por projeto/agente.
        </p>
      </section>
    </Shell>
  );
}
